355/*****
356Return true if this poly is still valid, i.e., none of its nodes have moved
357and all the other things that made this a polygon of the tree remain true.
358*****/
359bool tmPoly::CalcPolyIsValid() const
360{
361 // If it's a sub poly, it's always deemed to be valid (but if its owner
362 // dies, then it dies).
363 if (mIsSubPoly) return true;
364
365 // If the number of nodes has changed, e.g., through deletion of a node, then
366 // it's invalid.
367 size_t numRingNodes = mRingNodes.size();
368 if (mNodeLocs.size() != numRingNodes) {
369 return false;
370 }
371
372 // If any of the nodes have moved more than our tolerance, return that we're
373 // invalid.
374 for (size_t i = 0; i < numRingNodes; ++i) {
375 tmFloat moveDist = Mag(mNodeLocs[i] - mRingNodes[i]->mLoc);
376 if (moveDist > MoveTol()) {
377 return false;
378 }
379 }
380
381 // If any of its paths are no longer polygon paths, it's invalid.
382 for (size_t i = 0; i < mRingPaths.size(); ++i)
383 if (!mRingPaths[i]->IsPolygonPath()) {
384 return false;
385 }
386
387 // If it's nonconvex, it's invalid.
388 if (!CalcPolyIsConvex()) {
389 return false;
390 }
391
392 // If it encloses a leaf node that's not part of its ring, it's invalid.
393 tmArray<tmNode*> leafNodes;
394 mTree->GetLeafNodes(leafNodes);
395 if (CalcPolyEnclosesNode(leafNodes)) {
396 return false;
397 }
398
399 // If we passed all the tests, it's still valid.
400 return true;
401}

491 TMASSERT(mRidgePath)tmOnAssert(!!(mRidgePath), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 491, L"mRidgePath");
492 ridgePaths.push_back(mRidgePath);
493 ridgeNodes.push_back(backInsetNode);
494 }
495 break;
496 }
497 default: {
498 // 3 or more distinct inset nodes; get the inner bits from the subpoly.
499 tmNode* frontInsetNode = mInsetNodes[frontOffset];
500 tmNode* backInsetNode = mInsetNodes[backOffset];
501 if (frontInsetNode == backInsetNode) {
502 // if they're the same node, we don't need to go to a subpoly; we've
503 // reached the peak of this ridgeline. Add it in, and we can start
504 // descending.
505 ridgeNodes.push_back(frontInsetNode);
506 }
507 else {
508 // But if they're not the same node, we have to get the missing bits
509 // from the subpoly. Here's where the recursion kicks in. We find the
510 // subpoly whose ring path is the inset version of the path in
511 // question. There should be only one.
512 tmPoly* subPoly = 0;
513 for (size_t i = 0; i < mOwnedPolys.size(); ++i) {
514 tmPoly* thePoly = mOwnedPolys[i];
515 if (!thePoly->mRingNodes.contains(frontInsetNode) ||
516 !thePoly->mRingNodes.contains(backInsetNode)) continue;
517 subPoly = thePoly;
518 }
519 TMASSERT(subPoly)tmOnAssert(!!(subPoly), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 519, L"subPoly");
520 subPoly->GetRidgelineNodesAndPaths(frontInsetNode, backInsetNode,
521 ridgeNodes, ridgePaths);
522 }
523 }
524 }
525 // Finally, add the back node and spoke path to the list.
526 ridgePaths.push_back(mSpokePaths[backOffset]);
527 ridgeNodes.push_back(backNode);
528 // There should always be one more node than path.
529 TMASSERT(ridgeNodes.size() == ridgePaths.size() + 1)tmOnAssert(!!(ridgeNodes.size() == ridgePaths.size() + 1), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 529, L"ridgeNodes.size() == ridgePaths.size() + 1");
530
531#ifdef TMDEBUG
532 // Make sure each node and path appear only once in their respective lists.
533 for (size_t i = 0; i < ridgeNodes.size(); ++i) {
534 TMASSERT(ridgeNodes.count(ridgeNodes[i]) == 1)tmOnAssert(!!(ridgeNodes.count(ridgeNodes[i]) == 1), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 534, L"ridgeNodes.count(ridgeNodes[i]) == 1");
535 }
536 for (size_t i = 0; i < ridgePaths.size(); ++i) {
537 TMASSERT(ridgePaths.count(ridgePaths[i]) == 1)tmOnAssert(!!(ridgePaths.count(ridgePaths[i]) == 1), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 537, L"ridgePaths.count(ridgePaths[i]) == 1");
538 }
539#endif // TMDEBUG
540
541}

544/*****
545Initialize a ridge vertex to be sorted by passing with the front and back
546vertices of the path that defines the sorting order. That order is in CW order
547around the midpoint of the path. We compute the angle, mapping the front point
548to -pi/2 radians and the back point to +pi/2; they will then be sorted by
549angle. We use angle sorting rather than dot product along the sorting path
550because the vertices at each end of 90 degree ridge creases have the same dot
551product and would be ordered by numerical roundoff.
552*****/
553tmPoly::SortableRidgeVertex::SortableRidgeVertex(tmVertex* aVertex,
554 const tmPoint& frontPt, const tmPoint& backPt) :
555 mVertex(aVertex)
556{
557 tmPoint pu = Normalize(backPt - frontPt);
558 tmPoint pv = RotateCCW90(pu);
559 tmPoint dm = 0.5 * (backPt + frontPt);
560 tmPoint dp = aVertex->GetLoc() - dm;
561 tmFloat du = Inner(dp, pu);
562 tmFloat dv = Inner(dp, pv);
563 mSortValue = atan2(du, dv);
564}
565
566
567/*****
568Sorting function
569*****/
570bool tmPoly::SortableRidgeVertex::operator< (const SortableRidgeVertex& rhs)
571 const
572{
573 return mSortValue < rhs.mSortValue;
574}

577/*****
578Construct a sorted list of the vertices along the ridgeline above the path that
579connects the two given nodes. We specify the nodes, rather than just passing a
580tmPath*, because we need the nodes properly ordered for the call to
581BuildRidgelineNodesAndPaths(); frontNode is not
582necessarily thePath->mNodes.front(). This is used in two places in
583BuildPolyContents(). Note the use of struct SortableRidgeVertex for sorting
584vertices.
585*****/
586void tmPoly::GetRidgelineVertices(tmNode* frontNode, tmNode* backNode,
587 tmArray<tmVertex*>& ridgeVertices)
588{
589 // ridgeVertices should initially be empty.
590 TMASSERT(ridgeVertices.empty())tmOnAssert(!!(ridgeVertices.empty()), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 590, L"ridgeVertices.empty()");
591
592 // Get the ridgeline nodes and paths. Note that ridgeNodes will be ordered
593 // along the ridgeline from frontNode to backNode.
594 tmArray<tmNode*> ridgeNodes;
595 tmArray<tmPath*> ridgePaths;
596 GetRidgelineNodesAndPaths(frontNode, backNode, ridgeNodes, ridgePaths);
597
598 // Prepare to associate a sorting value with each of the vertices in the
599 // ridgeline. Although ridgeNodes is ordered, we don't accumulate vertices
600 // in order so we'll have to go back and sort them once we've accumulated
601 // them all based on the sorting value we've associated with each vertex.
602 tmArray<SortableRidgeVertex> vdlist;
603 tmPoint p1 = frontNode->mLoc;
604 tmPoint p2 = backNode->mLoc;
605 tmPoint pu = Normalize(p2 - p1);
606
607 // Accumulate all node-owned vertices in the ridgeline. If there is any
608 // junction node that doesn't yet have a vertex, give it one; all junction
609 // vertices should have vertices, but ridgeline peaks above inactive border
610 // paths might have escaped so far.
611 for (size_t i = 0; i < ridgeNodes.size(); ++i) {
612 tmNode* ridgeNode = ridgeNodes[i];
613 if (ridgeNode->IsJunctionNode()) ridgeNode->GetOrMakeVertexSelf();
614 tmVertex* ridgeVertex = ridgeNode->GetVertex();
615 if (ridgeVertex)
616 vdlist.push_back(SortableRidgeVertex(ridgeVertex, p1, p2));
617 }
618
619 // Accumulate all path-owned vertices in the ridgeline
620 for (size_t i = 0; i < ridgePaths.size(); ++i) {
621 tmPath* ridgePath = ridgePaths[i];
622 for (size_t j = 0; j < ridgePath->mOwnedVertices.size(); ++j) {
623 tmVertex* ridgeVertex = ridgePath->mOwnedVertices[j];
624 vdlist.push_back(SortableRidgeVertex(ridgeVertex,p1, p2));
625 }
626 }
627
628 // Sort all the entries according to their distance along the path.
629 sort(vdlist.begin(), vdlist.end());
630
631 // Finally, copy the ordered list into the return array.
632 ridgeVertices.reserve(vdlist.size());
633 for (size_t i = 0; i < vdlist.size(); ++i)
634 ridgeVertices.push_back(vdlist[i].mVertex);
635
636 // Make sure we haven't double-listed any vertices.
637#ifdef TMDEBUG
638 for (size_t i = 0; i < ridgeVertices.size(); ++i) {
639 TMASSERT(ridgeVertices.count(ridgeVertices[i]) == 1)tmOnAssert(!!(ridgeVertices.count(ridgeVertices[i]) == 1), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 639, L"ridgeVertices.count(ridgeVertices[i]) == 1");
640 }
641#endif // TMDEBUG
642}

655
656/*****
657Build the contents of this tmPoly and its subPolys. This routine gets called
658after a tmPoly has been created with at least three sides. Note that subpolys
659get killed automatically when this object is destroyed because it owns them
660through mOwnedPolys. This routine may safely be called for an already-built
661tmPoly (nothing will happen).
662*****/
663void tmPoly::BuildPolyContents()
664{
665 // If this tmPoly already contains any inset nodes, it's already been built
666 // and we can stop here.
667 if (HasPolyContents()) return;
668
669 // We can assume that mRingNodes and mRingPaths are created and valid.
670 // mRingNodes is an ordered list of nodes running around the outside of the
671 // tmPoly. mRingPaths is an ordered list of Paths running around the outside
672 // of the tmPoly.
673
674#ifdef TMDEBUG
675 {
676 // To avoid infinite recursion (and the ickiness that entails), we need to
677 // make sure that if this is a subpoly, it's smaller than its owner.
678 tmPoly* aPoly = GetOwnerAsPoly();
679 if (aPoly) {
680 TMASSERT(mRingNodes.size() < aPoly->mRingNodes.size())tmOnAssert(!!(mRingNodes.size() < aPoly->mRingNodes.size
()), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp", 680, L"mRingNodes.size() < aPoly->mRingNodes.size()"
);
681 }
682 }
683#endif // TMDEBUG
684
685 ///////////////////////////////////
686 // Construction of subpolys
687 ///////////////////////////////////
688
689 tmNode* aNode;
690 tmPoint p;
691 size_t nn = mRingNodes.size();
692 TMASSERT(mRingNodes.size() >= 3)tmOnAssert(!!(mRingNodes.size() >= 3), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 692, L"mRingNodes.size() >= 3");
693 if (mRingNodes.size() == 3) {
694 // If there are 3 sides, we don't need to go through the insetting
695 // calculation; we can just create a single inset tmNode and map each of
696 // the outer ring nodes onto it.
697 tmPoint& p1 = mRingNodes[0]->mLoc;
698 tmPoint& p2 = mRingNodes[1]->mLoc;
699 tmPoint& p3 = mRingNodes[2]->mLoc;
700 p = Incenter(p1, p2, p3);
701 aNode = new tmNode(mTree, this, p);
702 aNode->mIsJunctionNode = true;
703 aNode->mElevation = mRingNodes.front()->mElevation + Inradius(p1, p2, p3);
704
705 mInsetNodes.push_back(aNode); // all three nodes map to the inset tmNode
706 mInsetNodes.push_back(aNode);
707 mInsetNodes.push_back(aNode);
708
709 mSpokePaths.push_back(new tmPath(this, mRingNodes[0], aNode));
710 mSpokePaths.push_back(new tmPath(this, mRingNodes[1], aNode));
711 mSpokePaths.push_back(new tmPath(this, mRingNodes[2], aNode));
712 }
713 else {
714 // If there are 4 or more sides, then we need to inset the tmPoly and break
715 // it up into one or more smaller Polys. This is a much more complicated
716 // sequence of actions!
717
718 //**********************
719 // Part I: find the inset distance that produces a reduced polygon
720 //**********************
721
722 // Construct tables of vectors used in the insetting process
723
724 vector<tmPoint> r(nn); // bisector vector at each corner
725 vector<tmPoint> rp(nn); // normalized vector to previous corner
726 vector<tmPoint> rn(nn); // normalized vector to next corner
727 vector<tmFloat> mr(nn); // magnitude of projection of r along previous side
728
729 size_t i, j;
730 for (i = 0; i < nn; ++i) {
731 // get offsets of previous (ip) and next (in) corner, including
732 // wrap-around effects.
733 size_t ip = size_t((i + nn - 1) % nn);
734 size_t in = size_t((i + 1) % nn);
735
736 tmPoint nip = mRingNodes[ip]->mLoc;
737 tmPoint nii = mRingNodes[i]->mLoc;
738 tmPoint nin = mRingNodes[in]->mLoc;
739
740 // construct bisector and the magnitude of its projections along the
741 // previous and next sides. All quantities are normalized to a unit inset
742 // distance h.
743 rp[i] = Normalize(nip - nii); // vector to previous corner
744 rn[i] = Normalize(nin - nii); // vector to next corner
745 tmPoint bis = Normalize(RotateCCW90(rn[i] - rp[i])); // angle bisector
746 r[i] = bis / Inner(bis, RotateCCW90(rn[i])); // normalize to unit inset
747 mr[i] = Inner(r[i], rp[i]); // cotangent of bisected angle
748 }
749
750 // Now compute the maximum value of the inset distance h that satifies the
751 // reduced path condition for every path in the polygon. We handle RingPaths
752 // and CrossPaths differently. i and j are the indices of the corners of the
753 // path.
754 const tmFloat HMAX = 1.0e10;
755 tmFloat h = HMAX; // the best inset distance to use
756 for (i = 0; i < nn - 1; ++i)
757 for (j = i + 1; j < nn; ++j) {
758
759 // If the angle bisectors are parallel and are pointing the same
760 // direction, there's no solution, so go on to the next corner.
761 if (AreParallel(r[i], r[j]) && (Inner(r[i], r[j]) > 0)) continue;
762
763 // Get coordinates of the two nodes that we're checking
764 tmPoint ni = mRingNodes[i]->mLoc;
765 tmPoint nj = mRingNodes[j]->mLoc;
766
767 // for paths between adjacent nodes, we'll use the intersection of
768 // the bisectors to determine the maximum inset.
769 if ((j == i + 1) || ((i == 0) && (j == nn - 1))) {
770
771 tmPoint bi;
772 GetLineIntersection(ni, r[i], nj, r[j], bi);
773 tmFloat h1 = Inner((bi - ni), RotateCCW90(rn[i]));
774
775 // If the computed inset (h1) is smaller than the current inset (h),
776 // reduce the current inset. Something to check for: if the angle
777 // bisectors are barely NOT parallel, then we might have a very large
778 // positive or negative h, which is bogus. We don't worry about
779 // positive values, but negative values would cause big problems. So
780 // we'll ignore negative values of h1 when we do the comparison.
781 if ((h1 > 0) && (h > h1)) h = h1;
782 }
783 else {
784 // for paths between nonadjacent nodes, we'll compute the inset
785 // distance (using the cotangents of the base angles) that makes the
786 // given path active. If the computed inset distance is either complex
787 // or negative, there's no solution. We'll keep the smallest inset
788 // distance that we find.
789
790 // Note: an alternate way of structuring this would be to iterate
791 // through the mRingPaths and mCrossPaths separately; I wouldn't have
792 // to call FindPath. But I reference the tmNode information in the
793 // vectors by what amounts to the index of the mRingNode. So I do it
794 // this way. A better way, which I should implement in the future, is
795 // to put the vector data into scratch fields of the tmNode.
796
797 // Note that if the reduced path length comes out to be negative,
798 // we've found a spurious solution; so we have to detect and
799 // eliminate that case.
800 tmFloat lij = mPolyOwner->FindAnyPath(mRingNodes[i],
801 mRingNodes[j])->mMinPaperLength;
802 tmPoint u = ni - nj;
803 tmPoint v = r[i] - r[j];
804 tmFloat w = mr[i] + mr[j];
805 tmFloat a = Mag2(v) - pow(w, 2);
806 tmFloat b = Inner(u, v) + lij * w;
807 tmFloat c = Mag2(u) - pow(lij, 2);
808 tmFloat d = pow(b, 2) - a * c;
809 if (d < 0) continue; // both solutions are complex
810
811 tmFloat h1 = (-b + sqrt(d)) / a; // trial solution for inset distance
812 tmFloat lijp = lij - h1 * (mr[i] + mr[j]); // reduced path length
813 if ((lijp > 0) && (h1 > 0) && (h > h1)) h = h1;
814
815 h1 = (-b - sqrt(d)) / a; // other trial solution
816 lijp = lij - h1 * (mr[i] + mr[j]); // reduced path length
817 if ((lijp > 0) && (h1 > 0) && (h > h1)) h = h1;
818 }
819 }
820
821 // If we didn't find an acceptable value of the inset, then something's
822 // wrong.
823 TMASSERT(h != HMAX)tmOnAssert(!!(h != HMAX), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 823, L"h != HMAX");
824
825 // At this point, the variable h now holds the valid inset distance.
826
827 //**********************
828 // Part II: build a list of inset nodes and inset paths. These become the
829 // mOwnedNodes and mOwnedPaths of this poly.
830 //**********************
831
832 // We'll go through the list of corners and inset each corner, making a new
833 // list of distinct inset corners. In many cases, two or more corners have
834 // been inset to the same point (i.e., there's a net reduction in the
835 // number of corners); so we'll also construct a list of references mapping
836 // old corners to new, inset corners. We create new nodes (these are
837 // sub-nodes owned by this tmPoly) as needed; the member list mInsetNodes
838 // maintains the mapping, while a list of the distinct subnodes is provided
839 // by mOwnedNodes.
840 for (i = 0; i < nn; ++i) {
841 p = mRingNodes[i]->mLoc + h * r[i]; // inset corner location
842 mInsetNodes.push_back(GetOrMakeInsetNode(p));
843 }
844 size_t numOwnedNodes = mOwnedNodes.size();
845
846 // Set elevation of newly-created nodes
847 for (size_t i = 0; i < numOwnedNodes; ++i)
848 mOwnedNodes[i]->mElevation = mRingNodes.front()->mElevation + h;
849
850#ifdef TMDEBUG
851 // As an error check, make sure that all of the inset corners are still
852 // enclosed by the outer ring.
853 bool goodEnclosure = true;
854 for (i = 0; i < numOwnedNodes; ++i) {
855 goodEnclosure &= ConvexEncloses(mOwnedNodes[i]->mLoc);
856 if (!goodEnclosure) {
857 // If enclosure was bad, let's dump some diagnostic information to
858 // files.
859 DumpToFileAsText("BAD_BUILD_SUBPOLY.txt");
860 DumpToFileAsMath("BAD_BUILD_SUBPOLY.ma");
861
862 // break on this call to find the offending node
863 ConvexEncloses(mOwnedNodes[i]->mLoc);
864 TMFAIL("In tmPoly::BuildPolyContents() inset corner was not enclosed")tmOnAssert(0, L"../Source/tmModel/tmTreeClasses/tmPoly.cpp", 864
, L"fail", L"In tmPoly::BuildPolyContents() inset corner was not enclosed"
);
865 }
866 }
867#endif // TMDEBUG
868
869 // Now we have a valid inset. The next action depends on how many distinct
870 // inset nodes we have.
871 if ((numOwnedNodes == 1) || (numOwnedNodes == 2)) {
872 // if 1 or 2 nodes, we're done building molecules. The last thing we do
873 // in preparation for building the Creases is to create Paths for the
874 // spokes of the reduction.
875 for (i = 0; i < nn; ++i)
876 mSpokePaths.push_back(new tmPath(this, mRingNodes[i], mInsetNodes[i]));
877 // If there are exactly two distinct inset nodes, we create a tmPath
878 // corresponding to the ridge.
879 if (numOwnedNodes == 2)
880 mRidgePath = new tmPath(this, mOwnedNodes[0], mOwnedNodes[1]);
881 }
882 else {
883 // If there are 3 or more distinct inset nodes, this tmPoly will have at
884 // least one sub-tmPoly. We build a new complete set of reduced paths
885 // from which we can construct new polys.
886
887 // construct the distinct inset paths that link the inset nodes. These
888 // will consist of ring paths and cross paths. Since multiple outer nodes
889 // may map to a single inset node, multiple outset paths can potentially
890 // map to the same inset path. For each ring path of the inset poly, we
891 // want its outset path to be a ring path of the outset poly; that is, a
892 // path whose nodes are consecutive members of mRingNodes. So, we'll go
893 // through every distinct pair of nodes in mRingNodes, but choosing an
894 // order such that we go through all the consecutive pairs of mRingNodes
895 // before eventually trying every possible pair.
896 bool madeActiveCrossPath = false;
897 for (size_t dij = 1; dij < nn; ++dij)
898 for (size_t i = 0; i <= nn - dij; ++i) {
899 size_t j = (i + dij) % nn;
900 tmNode* ni = mRingNodes[i];
901 tmNode* nj = mRingNodes[j];
902 tmNode* rni = mInsetNodes[i];
903 tmNode* rnj = mInsetNodes[j];
904
905 // if both outer nodes map to the same inner tmNode, go on to the next
906 // pair.
907 if (rni == rnj) continue;
908
909 // If a path already exists between the two inset nodes go on to the
910 // next pair.
911 if (FindLeafPath(rni, rnj)) continue;
912
913 // if we didn't find it, need to create a new path.
914 tmPath* outsetPath = mPolyOwner->FindAnyPath(ni, nj);
915 TMASSERT(outsetPath)tmOnAssert(!!(outsetPath), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 915, L"outsetPath");
916 tmFloat iReduction = h * mr[i];
917 tmFloat jReduction = h * mr[j];
918
919 // When we create the new path, we want it in the same orientation as
920 // the path from which it is inset, so order of rni, rnj is important.
921 tmPath* thePath;
922 if (outsetPath->mNodes.front() == ni) {
923 // rni should be front node
924 thePath = new tmPath(this, rni, rnj);
925 thePath->mFrontReduction = iReduction;
926 thePath->mBackReduction = jReduction;
927 }
928 else {
929 // rnj should be front node
930 thePath = new tmPath(this, rnj, rni);
931 thePath->mFrontReduction = jReduction;
932 thePath->mBackReduction = iReduction;
933 }
934
935 // Connect it to the ring nodes it was created from
936 rni->mLeafPaths.push_back(thePath);
937 rnj->mLeafPaths.push_back(thePath);
938
939 // Set length-related variables
940 thePath->mOutsetPath = outsetPath;
941 thePath->mMinPaperLength = outsetPath->mMinPaperLength -
942 (thePath->mFrontReduction + thePath->mBackReduction);
943 thePath->mActPaperLength = Mag((rni->mLoc) - (rnj->mLoc));
944 thePath->mMinTreeLength = thePath->mMinPaperLength / mTree->mScale;
945 thePath->mActTreeLength = thePath->mActPaperLength / mTree->mScale;
946
947 // Set active status; we're definitely active if the path we're inset
948 // from is active, but could be active even if our outset path is not.
949 thePath->mIsActivePath = outsetPath->IsActivePath();
950
951 // if the reduced path length is equal to its physical length, it's
952 // active whether its outset path was active or not. (If it's outset
953 // path was not active and this one is, then it's an active cross
954 // path.)
955 thePath->mIsActivePath |= thePath->TestIsActive(
956 thePath->mActPaperLength, thePath->mMinPaperLength);
957
958 // if the nodes are consecutive, the inset path is a border path and
959 // is therefore also a polygon path. If it isn't a border path, note
960 // whether it's active.
961 if (dij == 1) thePath->mIsBorderPath = true;
962 else madeActiveCrossPath |= thePath->IsActivePath();
963
964 // If it's either active or border, it's a polygon path
965 thePath->mIsPolygonPath =
966 thePath->IsActivePath() || thePath->IsBorderPath();
967 }
968
969#ifdef TMDEBUG
970 // count the number of polygon paths we created
971 size_t numPolygonPaths = 0;
972 for (size_t i = 0; i < mOwnedPaths.size(); ++i)
973 if (mOwnedPaths[i]->IsPolygonPath()) ++numPolygonPaths;
974 // Another debugging check: we must have either reduced the number of
975 // inset nodes or created an active CrossPath. If we didn't do either,
976 // something is wrong.
977 TMASSERT(madeActiveCrossPath || (mOwnedNodes.size() < mRingNodes.size()))tmOnAssert(!!(madeActiveCrossPath || (mOwnedNodes.size() <
mRingNodes.size())), L"../Source/tmModel/tmTreeClasses/tmPoly.cpp"
, 977, L"madeActiveCrossPath || (mOwnedNodes.size() < mRingNodes.size())"
);
978#endif
979
980 // Now create sub-polygons from the network of paths. We pass the inset
981 // nodes as the list of border nodes, from which the centroid of the
982 // border nodes (which is also the convex hull) is computed.
983 BuildPolysFromPaths(mOwnedPaths, mInsetNodes);
984
985 // For the newly-built polys, call their BuildSubPolys routines to
986 // recursively build all levels.
987 tmPoly* aPoly;
988 tmArrayIterator<tmPoly*> iOwnedPolys(mOwnedPolys);
989 while (iOwnedPolys.Next(&aPoly)) aPoly->BuildPolyContents();
990
991 // The last thing we do in preparation for building the Creases is to
992 // create Paths for the spokes of the reduction.
993 for (size_t i = 0; i < nn; ++i)
994 mSpokePaths.push_back(new tmPath(this, mRingNodes[i], mInsetNodes[i]));
995 } // End of branch with numOwnedNodes >= 3
996 } // End of branch with (mRingNodes.size() > 3)
997
998 /////////////////////////////////
999 // Build creases
1000 /////////////////////////////////
1001
1002 // Build the crease pattern within this poly (or, if it's a subpoly, build
1003 // certain portions of the crease pattern -- specifically, those parts
1004 // associated with gusset paths and creases. Note that all ridge and hinge
1005 // creases should end up owned by the outermost poly, even if we're creating
1006 // them from within a subpoly.
1007 for (size_t i = 0; i < nn; ++i) {
1008 tmNode* frontNode = mRingNodes[i];
1009 tmNode* backNode = mRingNodes[(i + 1) % nn];
1010 tmPath* thePath = mRingPaths[i];
1011
1012 // For ALL axial paths and gusset paths, build vertices along the path.
1013 if (thePath->IsAxialPath() || thePath->IsGussetPath())
1014 thePath->BuildSelfVertices();
1015
